  def test_gets
    o = OhEmbedr::OhEmbedr.new(:url => "http://vimeo.com/6382511")
    o.gets
    
    # Make sure we got an hash
    assert_equal o.hash.class, Hash
    
    # Make sure there's the right kinda stuff in there
    assert_equal o.hash["type"], "video"
  end
